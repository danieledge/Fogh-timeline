.stack3d {
  position: relative;
  width: min(900px, 100%);
  margin: 0 auto;
  aspect-ratio: 16 / 9;
  perspective: 1200px;
  perspective-origin: 50% 50%;
  overflow: visible; /* allow depth overflow */
  touch-action: pan-y; /* keep page scroll vertical */
  outline: none; /* we’ll set focus ring on active slide */
}

.stack3d > img {
  position: absolute; inset: 0;
  margin: auto;
  max-width: 70%;
  max-height: 80%;
  transform-style: preserve-3d;
  transition: transform 400ms ease, filter 300ms ease, opacity 300ms ease;
  will-change: transform;
  border-radius: 8px;
  box-shadow: 0 10px 30px rgba(0,0,0,.25);
  opacity: 0;               /* hidden until laid out */
  backface-visibility: hidden;
}

.stack3d > img.is-visible { opacity: 1; }

/* Focus/active styling */
.stack3d > img[data-active="true"] {
  outline: 2px solid currentColor;
  outline-offset: 3px;
  filter: none !important;
}

/* Optional controls */
